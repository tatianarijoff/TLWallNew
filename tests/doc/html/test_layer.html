Example Usage

from pytlwall.layer import Layer
import numpy as np

# Copper layer with frequencies
freq = np.logspace(3, 9, 100)  # 1 kHz to 1 GHz
copper = Layer(
    layer_type='CW',
    thick_m=0.001,           # 1 mm
    sigmaDC=5.96e7,          # Copper conductivity
    epsr=1.0,
    RQ=0.5e-6,               # 0.5 Œºm roughness
    freq_Hz=freq
)

# Get calculated properties
print(f"Skin depth at 1 GHz: {copper.delta[-1]*1e6:.2f} Œºm")
print(f"Surface resistance: {copper.RS[-1]:.4f} Œ©")
